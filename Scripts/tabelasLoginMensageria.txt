CREATE TABLE TB_LOGIN (
    ID              UNIQUEIDENTIFIER PRIMARY KEY NOT NULL,
    CPF             VARCHAR (11),
    NOME            VARCHAR (100),
    EMAIL           VARCHAR (200),
    SENHA           VARCHAR (100),
    DATACRIACAO     DATETIME,
    DATAATUALIZACAO DATETIME,
    ATIVADO         BIT
);

CREATE TABLE TB_CODIGOCADASTRO
(
	ID 		UNIQUEIDENTIFIER NOT NULL,
	IDLOGIN		UNIQUEIDENTIFIER
	CODIGO 		UNIQUEIDENTIFIER,
	DATACRIACAO 	DATETIME,
	DATAEXPIRACAO 	DATETIME

	CONSTRAINT fk_Login FOREIGN KEY (IDLOGIN) REFERENCES TB_LOGIN (ID), 
)
